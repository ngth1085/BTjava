public class cricle {

    public static final double DEFAULT_RADIUS = 8.8;

    private double radius;

    public cricle() {                   // 1st (default) Constructor
        this.radius = DEFAULT_RADIUS;
    }
    public cricle(double radius) {      // 2nd Constructor
        this.radius = radius;
    }
    public cricle (double radius,String color) { // 3rd Constructor
        this.radius = radius;
    }

    // The public getters and setters for the private variables
    public double getRadius() {
        return this.radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    // The toString() returns a String description of this instance
    public String toString() {
        return "Circle[radius=" + radius + "]";
    }

    // Return the area of this Circle
    public double getArea() {
        return radius * radius * Math.PI;
    }
